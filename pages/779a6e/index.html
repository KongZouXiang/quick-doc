# 使用

# ResultBody

注解Controller类或者方法上自动包装统一返回内容

# 实例

/**
 * 系统用户user表控制层
 *
 * @author makejava
 * @since 2022-03-22 16:43:23
 */
@ResultBody
@Tag(name = "系统用户")
@RestController
@RequestMapping(rootPath+"user")
public class UserController extends BaseController<User, UserService> {}
